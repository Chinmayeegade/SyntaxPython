MULTICOLOR HEXAGONAL SPIRAL IN TURTLE

import turtle
col = ("red","orange","yellow","green","blue","indigo","violet")
t = turtle.Turtle()
screen = turtle.Screen()
screen.bgcolor("white")
t.speed(100)
t.shape("square")


for i in range(300):
    t.color(col[i%7])
    t.forward(i*1.5)
    t.left(70)
    t.width(5)
