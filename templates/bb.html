BOUNCING IMAGE IN PYGAME

import sys
import pygame
from pygame.locals import *

pygame.init()
surface = pygame.display.set_mode((640,480))

image = pygame.image.load("download.jpg")
#download.jpg in the image stored in device
#download.jpg in the image stored in device
image_rect = image.get_rect()
time = pygame.time.Clock()

width = surface.get_width()
height = surface.get_height()

speed = [4,4]
while True:
    for ev in pygame.event.get():
        if ev.type == QUIT:
            pygame.quit()
    time.tick(60)
    surface.fill((0,0,0))
    image_rect = image_rect.move(speed)
    if image_rect.left < 0 or image_rect.right > width:
        speed[0] = -speed[0]
    if image_rect.top<0 or image_rect.bottom > height:
        speed[1] = -speed[1]
    surface.blit(image,image_rect)
    pygame.display.flip()
