RANDOM DICE ROLL IN PYTHON

import random
from random import choice
x = int(input("How many dice do you want to roll? "))
n = int(input("How many times do you want to roll the die/dice? "))
option = [1,2,3,4,5,6]
series = []
for i in range(1,n+1,1):
    for c in range(0,x,1):
     a = random.choice(option)
     series.append(a)
    print("Roll",i,"=",sum(series))
    input("Press 'Enter' key")
    series.clear()
