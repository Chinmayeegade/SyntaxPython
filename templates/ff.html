PIE CHART USING MATPLOTLIB

import matplotlib
import matplotlib.pyplot as pie
divisions = int(input("Enter the number of data categories: "))
total = int(input("Enter the total value of the data: "))
data = []
values = []
for n in range(1,divisions+1,1):
    print("Enter the title of data category number",n,": ")
    category = input()
    data.append(category)
    print("Enter the value of",data[n-1],": ")
    Input = int(input())
    percent = (100*Input)/total
    values.append(percent)
pie.pie(x=values, labels=data)
pie.show()
